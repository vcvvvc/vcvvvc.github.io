V1


############ 交叉编译
CROSS_COMPILE=aarch64-himix100-linux-
CC=$(CROSS_COMPILE)g++

# 当前路径
PWD := $(shell pwd)

#库
LIB += -lpthread -lsqlite3

#参数
CFLAG= -D_GNU_SOURCE -D__USE_XOPEN -g

#.cpp源文件
SRCS=$(wildcard ./src/*.cpp)

#头文件
INC=-I$(PWD)/include

#中间文件
OBJS=$(SRCS:.cpp=.o)

#二进制文件
BIN=maincpp

#start，表示开始执行
start:$(OBJS)
	$(CC) -o $(BIN) $(OBJS) $(CFLAG) $(LIB)

.cpp.o:
	$(CC) $(INC)  -o $@ -c $< $(CFLAB) $(LIB)

.PHONY : clean
clean:
	rm -f ${BIN} ./src/*.o

# Makefile
cc=gcc

LDFLAGS=-lncurses

SRCS := $(wildcard *.c)
TARGET := $(SRCS:%.c=%)

$(TARGET):$(SRCS)
	$(cc) $(LDFLAGS) $(SRCS) -o $(TARGET)

clean:
	rm $(TARGET)
